1    package com.shopapp.database;
2    
3    import com.shopapp.models.CartItem;
4    import com.shopapp.models.Invoice;
5    import java.sql.*;
6    import java.util.List;
7    
8    public class InvoiceDAO {
9    
10       public void createInvoice(String custName, String custPhone, String gst, List<CartItem> items, double totalAmount) throws SQLException {
11           // Simple logic: totalAmount is Final Amount (inclusive of tax for now, unless we change calculation)
12           // Let's assume passed total is final.
13           
14           String sqlInv = "INSERT INTO invoices (customer_name, customer_contact, gst_number, total_amount, gst_total, final_amount) VALUES (?, ?, ?, ?, ?, ?)";
15           String sqlItem = "INSERT INTO invoice_items (invoice_id, product_id, product_name, quantity, rate, gst_percent, total) VALUES (?, ?, ?, ?, ?, ?, ?)";
16   
17           try (Connection conn = DatabaseManager.getConnection()) {
18               // Start Transaction
19               conn.setAutoCommit(false);
20               
21               try {
22                   // 1. Insert Invoice
23                   int invoiceId = 0;
24                   try (PreparedStatement pstmt = conn.prepareStatement(sqlInv, Statement.RETURN_GENERATED_KEYS)) {
25                       pstmt.setString(1, custName);
26                       pstmt.setString(2, custPhone);
27                       pstmt.setString(3, gst);
28                       pstmt.setDouble(4, totalAmount); // Base
29                       pstmt.setDouble(5, 0); // No GST calc yet in logic
30                       pstmt.setDouble(6, totalAmount); // Final
31                       pstmt.executeUpdate();
32                       
33                       ResultSet rs = pstmt.getGeneratedKeys();
34                       if (rs.next()) invoiceId = rs.getInt(1);
35                       else throw new SQLException("Failed to create invoice");
36                   }
37   
38                   // 2. Insert Items
39                   try (PreparedStatement pstmt = conn.prepareStatement(sqlItem)) {
40                       for (CartItem item : items) {
41                           pstmt.setInt(1, invoiceId);
42                           pstmt.setInt(2, item.getProduct().getId());
43                           pstmt.setString(3, item.getProduct().getName());
44                           pstmt.setInt(4, item.getQuantity());
45                           pstmt.setDouble(5, item.getProduct().getPricePerUnit());
46                           pstmt.setDouble(6, 0); // GST %
47                           pstmt.setDouble(7, item.getTotal());
48                           pstmt.addBatch();
49                       }
50                       pstmt.executeBatch();
51                   }
52   
53                   conn.commit();
54               } catch (SQLException e) {
55                   conn.rollback();
56                   throw e;
57               } finally {
58                   conn.setAutoCommit(true);
59               }
60           }
61       }
62   }
