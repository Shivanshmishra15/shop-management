11       public void addExpense(Expense exc) throws SQLException {
12           String sql = "INSERT INTO expenses (category, description, amount) VALUES (?, ?, ?)";
13           try (Connection conn = DatabaseManager.getConnection();
14                PreparedStatement pstmt = conn.prepareStatement(sql)) {
15               pstmt.setString(1, exc.getCategory());
16               pstmt.setString(2, exc.getDescription());
17               pstmt.setDouble(3, exc.getAmount());
18               pstmt.executeUpdate();
19           }
20       }
21   
22       public List<Expense> getAllExpenses() throws SQLException {
23           List<Expense> list = new ArrayList<>();
24           String sql = "SELECT * FROM expenses ORDER BY expense_date DESC";
25           try (Connection conn = DatabaseManager.getConnection();
26                Statement stmt = conn.createStatement();
27                ResultSet rs = stmt.executeQuery(sql)) {
28               while (rs.next()) {
29                   LocalDateTime date;
30                   try { date = LocalDateTime.parse(rs.getString("expense_date").replace(" ", "T")); }
31                   catch (Exception e) { date = LocalDateTime.now(); }
32   
33                   list.add(new Expense(
34                       rs.getInt("id"),
35                       rs.getString("category"),
36                       rs.getString("description"),
37                       rs.getDouble("amount"),
38                       date
39                   ));
40               }
41           }
42           return list;
43       }
44   
45       public void deleteExpense(int id) throws SQLException {
46           try (Connection conn = DatabaseManager.getConnection();
47                PreparedStatement pstmt = conn.prepareStatement("DELETE FROM expenses WHERE id=?")) {
48               pstmt.setInt(1, id);
49               pstmt.executeUpdate();
50           }
51       }
52   }
