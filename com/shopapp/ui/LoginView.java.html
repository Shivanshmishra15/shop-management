1    package com.shopapp.ui;
2    
3    import com.shopapp.database.UserDAO;
4    import com.shopapp.models.User;
5    import com.shopapp.Main;
6    import javafx.geometry.Insets;
7    import javafx.geometry.Pos;
8    import javafx.scene.Scene;
9    import javafx.scene.control.*;
10   import javafx.scene.layout.VBox;
11   import javafx.stage.Stage;
12   
13   import java.util.Objects;
14   import java.util.Optional;
15   
16   public class LoginView extends VBox {
17   
18       private final UserDAO userDAO;
19       private final Stage stage;
20   
21       public LoginView(Stage stage) {
22           this.stage = stage;
23           this.userDAO = new UserDAO();
24           
25           setAlignment(Pos.CENTER);
26           setSpacing(20);
27           setPadding(new Insets(40));
28           setStyle("-fx-background-color: #ffffff;"); // Light theme background
29   
30           // Logo / Title
31           Label lblTitle = new Label("Shop Manager");
32           lblTitle.setStyle("-fx-font-size: 32px; -fx-font-weight: bold; -fx-text-fill: #38bdf8;");
33           
34           Label lblSubtitle = new Label("Sign in to continue");
35           lblSubtitle.setStyle("-fx-text-fill: #64748b; -fx-font-size: 16px;");
36   
37           // Form
38           VBox form = new VBox(15);
39           form.setMaxWidth(350);
40           form.setAlignment(Pos.CENTER_LEFT);
41   
42           Label lblUser = new Label("Username");
43           lblUser.setStyle("-fx-text-fill: #1e293b; -fx-font-size: 14px;");
44           TextField txtUser = new TextField();
45           txtUser.setPromptText("Enter username (e.g., admin)");
46           txtUser.getStyleClass().add("text-field");
47   
48           Label lblPass = new Label("Password");
49           lblPass.setStyle("-fx-text-fill: #1e293b; -fx-font-size: 14px;");
50           PasswordField txtPass = new PasswordField();
51           txtPass.setPromptText("Enter password");
52           txtPass.getStyleClass().add("text-field");
53   
54           Button btnLogin = new Button("Login");
55           btnLogin.getStyleClass().add("button-primary");
56           btnLogin.setMaxWidth(Double.MAX_VALUE);
57           btnLogin.setStyle("-fx-font-size: 14px; -fx-padding: 10 20;");
58           
59           Label lblError = new Label();
60           lblError.setStyle("-fx-text-fill: #ef4444;");
61           lblError.setVisible(false);
62   
63           // Login Logic
64           btnLogin.setOnAction(e -> {
65               String u = txtUser.getText();
66               String p = txtPass.getText();
67               
68               if (u.isEmpty() || p.isEmpty()) {
69                   lblError.setText("Please enter both username and password.");
70                   lblError.setVisible(true);
71                   return;
72               }
73   
74               Optional<User> user = userDAO.authenticate(u, p);
75               if (user.isPresent()) {
76                   launchMainApp(user.get());
77               } else {
78                   lblError.setText("Invalid credentials.");
79                   lblError.setVisible(true);
80               }
81           });
82   
83           form.getChildren().addAll(lblUser, txtUser, lblPass, txtPass, lblError, btnLogin);
84           getChildren().addAll(lblTitle, lblSubtitle, form);
85       }
86   
87       private void launchMainApp(User user) {
88           // Transition to Main App
89           MainLayout mainRoot = new MainLayout(user); // Pass user to layout
90           Scene scene = new Scene(mainRoot, 1280, 800);
91           scene.getStylesheets().add(Objects.requireNonNull(Main.class.getResource("/styles.css")).toExternalForm());
92           
93           stage.setScene(scene);
94           stage.centerOnScreen();
95       }
96   }
