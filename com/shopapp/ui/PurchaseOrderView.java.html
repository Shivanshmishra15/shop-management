56       private VBox createOrderView() {

140      private VBox createHistoryView() {
141          VBox root = new VBox(20);
142          root.setPadding(new Insets(20));
143          
144          historyTable = new TableView<>();
145          
146          TableColumn<PurchaseOrder, Integer> idCol = new TableColumn<>("PO #");
147          idCol.setCellValueFactory(new PropertyValueFactory<>("id"));
148          
149          TableColumn<PurchaseOrder, String> supCol = new TableColumn<>("Supplier");
150          supCol.setCellValueFactory(new PropertyValueFactory<>("supplierName"));
151          
152          TableColumn<PurchaseOrder, String> dateCol = new TableColumn<>("Date");

158          amtCol.setCellValueFactory(new PropertyValueFactory<>("totalAmount"));
159          
160          TableColumn<PurchaseOrder, String> statCol = new TableColumn<>("Status");
161          statCol.setCellValueFactory(new PropertyValueFactory<>("status"));
162  
163          historyTable.getColumns().addAll(idCol, supCol, dateCol, amtCol, statCol);

180      }

182      private void loadHistory() {
183          try {
184              historyTable.setItems(FXCollections.observableArrayList(poDAO.getAllOrders()));
185          } catch (Exception e) { e.printStackTrace(); }
186      }

188      private void addToOrder() {
189          Product p = productTable.getSelectionModel().getSelectedItem();
190          if (p == null) return;
191          
192          // Simple input dialog for Qty
193          TextInputDialog dialog = new TextInputDialog("10");
194          dialog.setTitle("Quantity");
195          dialog.setHeaderText("Enter quantity for " + p.getName());
196          dialog.setContentText("Quantity:");
197          
198          dialog.showAndWait().ifPresent(qtyStr -> {
199              try {
200                  int qty = Integer.parseInt(qtyStr);
201                  // Assume Cost Price is same as pricePerUnit for simplicity, or ask user?
202                  // Let's ask for Cost Price too? Keeping it simple: take current price as default cost
203                  double cost = p.getPricePerUnit(); 
204                  
205                  PurchaseOrderItem item = new PurchaseOrderItem(0, 0, p.getId(), qty, cost);
206                  item.setProductName(p.getName());
207                  orderItems.add(item);
208                  updateTotal();
209              } catch (NumberFormatException e) { /* ignore */ }
210          });
211      }
212  
213      private void updateTotal() {
214          double sum = orderItems.stream().mapToDouble(PurchaseOrderItem::getTotal).sum();
215          lblTotal.setText(String.format("Total: %.2f", sum));
216      }
217  
218      private void submitOrder() {
219          Supplier s = cmbSupplier.getValue();
220          if (s == null) {
221              new Alert(Alert.AlertType.ERROR, "Select Supplier").show();
222              return;
223          }
224          if (orderItems.isEmpty()) {
225              new Alert(Alert.AlertType.ERROR, "Add items").show();
226              return;
227          }
228  
229          try {
230              PurchaseOrder po = new PurchaseOrder(0, s.getId(), null, "PENDING", 
231                  orderItems.stream().mapToDouble(PurchaseOrderItem::getTotal).sum());
232              po.setItems(new ArrayList<>(orderItems));
233              
234              poDAO.createPurchaseOrder(po);
235              
236              new Alert(Alert.AlertType.INFORMATION, "Order Created Successfully").show();
237              orderItems.clear();
238              updateTotal();
239          } catch (Exception e) {
240              e.printStackTrace();
241              new Alert(Alert.AlertType.ERROR, "Failed: " + e.getMessage()).show();
242          }
243      }
