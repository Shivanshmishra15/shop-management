1    package com.shopapp.ui;
2    
3    import com.shopapp.database.ProductDAO;
4    import javafx.geometry.Insets;
5    import javafx.scene.Cursor;
6    import javafx.scene.control.Label;
7    import javafx.scene.layout.*;
8    import java.util.function.Consumer;
9    
10   public class DashboardView extends VBox {
11   
12       private final ProductDAO productDAO;
13       private final Consumer<String> navigationCallback;
14   
15       public DashboardView(Consumer<String> navigationCallback) {
16           this.navigationCallback = navigationCallback;
17           this.productDAO = new ProductDAO();
18           setSpacing(25);
19           setPadding(new Insets(10));
20   
21           // Header
22           Label title = new Label("Dashboard");
23           title.getStyleClass().add("page-title");
24   
25           // Stats Grid
26           HBox statsContainer = new HBox(20);
27           statsContainer.getChildren().addAll(
28                   createStatCard("Total Products", "124", "blue", "INVENTORY"),
29                   createStatCard("Low Stock Alerts", "5", "red", "INVENTORY_LOW"),
30                   createStatCard("Inventory Value", "$12,450", "green", "REPORTS")
31           );
32   
33           // Recent Activity Section
34           VBox activitySection = new VBox(15);
35           activitySection.getStyleClass().add("card");
36           Label activityTitle = new Label("Recent Activity");
37           activityTitle.setStyle("-fx-font-size: 18px; -fx-text-fill: #1e293b; -fx-font-weight: bold;");
38           
39           Label activityPlaceholder = new Label("• New Product 'Wireless Mouse' added\n• Invoice #1024 created\n• Stock updated for 'HDMI Cable'");
40           activityPlaceholder.setStyle("-fx-text-fill: #475569; -fx-font-size: 14px; -fx-line-spacing: 8;");
41           
42           activitySection.getChildren().addAll(activityTitle, activityPlaceholder);
43   
44           getChildren().addAll(title, statsContainer, activitySection);
45       }
46   
47       private VBox createStatCard(String title, String value, String colorTheme, String targetView) {
48           VBox card = new VBox(10);
49           card.getStyleClass().add("card");
50           HBox.setHgrow(card, Priority.ALWAYS);
51           card.setMinWidth(200);
52           
53           // Interaction
54           card.setCursor(Cursor.HAND);
55           card.setOnMouseClicked(e -> navigationCallback.accept(targetView));
56           // Hover effect
57           card.setOnMouseEntered(e -> card.setStyle("-fx-background-color: #e0f2fe; -fx-background-radius: 16; -fx-effect: dropshadow(three-pass-box, rgba(14,165,233,0.3), 10, 0, 0, 0);"));
58           card.setOnMouseExited(e -> card.setStyle("-fx-background-radius: 16;")); // Reset to CSS default (roughly) - NOTE: Ideally toggle style class
59   
60           Label lblTitle = new Label(title);
61           lblTitle.setStyle("-fx-text-fill: #64748b; -fx-font-size: 14px; -fx-font-weight: bold;");
62   
63           Label lblValue = new Label(value);
64           lblValue.setStyle("-fx-text-fill: #1e293b; -fx-font-size: 32px; -fx-font-weight: bold;");
65   
66           Region indicator = new Region();
67           indicator.setMaxHeight(4);
68           indicator.setMinHeight(4);
69           String colorCode = switch (colorTheme) {
70               case "red" -> "#ef4444";
71               case "green" -> "#10b981";
72               default -> "#0ea5e9";
73           };
74           indicator.setStyle("-fx-background-color: " + colorCode + "; -fx-background-radius: 2;");
75   
76           card.getChildren().addAll(lblTitle, lblValue, indicator);
77           return card;
78       }
79   }
