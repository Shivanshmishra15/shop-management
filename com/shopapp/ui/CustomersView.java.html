98       private void setupTable() {
99           TableColumn<Customer, String> colName = new TableColumn<>("Name");
100          colName.setCellValueFactory(new PropertyValueFactory<>("name"));
101          
102          TableColumn<Customer, String> colPhone = new TableColumn<>("Phone");
103          colPhone.setCellValueFactory(new PropertyValueFactory<>("phone"));
104          
105          TableColumn<Customer, Integer> colPoints = new TableColumn<>("Points");
106          colPoints.setCellValueFactory(new PropertyValueFactory<>("loyaltyPoints"));
107  
108          table.getColumns().addAll(colName, colPhone, colPoints);
109          table.setColumnResizePolicy(TableView.CONSTRAINED_RESIZE_POLICY);
110      }
111  
112      private void loadData(String query) {
113          Task<List<Customer>> task = new Task<>() {
114              @Override
115              protected List<Customer> call() throws Exception {
116                  if (query == null || query.isEmpty()) {
117                      return customerDAO.getAllCustomers();
118                  } else {
119                      return customerDAO.searchCustomers(query);
120                  }
121              }
122          };
123          task.setOnSucceeded(e -> {
124              customerList = FXCollections.observableArrayList(task.getValue());
125              table.setItems(customerList);
126          });
127          new Thread(task).start();
128      }

130      private void saveCustomer() {
131          if (txtName.getText().isEmpty() || txtPhone.getText().isEmpty()) {
132              new Alert(Alert.AlertType.ERROR, "Name and Phone required").show();
133              return;
134          }
135  
136          try {
137              int points = 0;
138              try { points = Integer.parseInt(txtPoints.getText()); } catch (Exception e) {}
139              
140              Customer c = new Customer(
141                  selectedCustomer == null ? 0 : selectedCustomer.getId(),
142                  txtName.getText(),
143                  txtPhone.getText(),
144                  txtEmail.getText(),
145                  txtAddress.getText(),
146                  points,
147                  null
148              );
149  
150              if (selectedCustomer == null) {
151                  customerDAO.addCustomer(c);
152              } else {
153                  customerDAO.updateCustomer(c);
154              }
155              clearForm();
156              loadData("");
157          } catch (SQLException e) { e.printStackTrace(); }
158      }
