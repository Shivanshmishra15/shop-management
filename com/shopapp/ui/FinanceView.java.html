1    package com.shopapp.ui;
2    
3    import com.shopapp.database.ExpenseDAO;
4    import com.shopapp.database.FinanceDAO;
5    import com.shopapp.models.Expense;
6    import javafx.collections.FXCollections;
7    import javafx.geometry.Insets;
8    import javafx.geometry.Pos;
9    import javafx.scene.chart.*;
10   import javafx.scene.control.*;
11   import javafx.scene.control.cell.PropertyValueFactory;
12   import javafx.scene.layout.*;
13   import java.time.format.DateTimeFormatter;
14   
15   public class FinanceView extends BorderPane {
16   
17       private final FinanceDAO financeDAO;
18       private final ExpenseDAO expenseDAO;
19       private TableView<Expense> expenseTable;
20       
21       // Labels
22       private Label lblRevenue, lblCOGS, lblExpenses, lblProfit;
23   
24       public FinanceView() {
25           financeDAO = new FinanceDAO();
26           expenseDAO = new ExpenseDAO();
27           
28           getStyleClass().add("finance-view");
29           setPadding(new Insets(20));
30           
31           // Top: Dashboard Cards
32           HBox dashboard = createDashboardStats();
33           
34           // Center: Expense Manager
35           VBox expenseManager = createExpenseManager();
36           
37           setTop(dashboard);
38           setCenter(expenseManager);
39           
40           refreshStats();
41       }
42   
43       private HBox createDashboardStats() {
44           HBox box = new HBox(20);
45           box.setAlignment(Pos.CENTER);
46           box.setPadding(new Insets(0, 0, 20, 0));
47           
48           lblRevenue = createStatLabel("Total Revenue", "0.00", "#10b981");
49           lblCOGS = createStatLabel("PO Cost (COGS)", "0.00", "#f59e0b");
50           lblExpenses = createStatLabel("Op. Expenses", "0.00", "#f43f5e");
51           lblProfit = createStatLabel("Net Profit", "0.00", "#3b82f6");
52           
53           box.getChildren().addAll(lblRevenue.getParent(), lblCOGS.getParent(), lblExpenses.getParent(), lblProfit.getParent());
54           return box;
55       }
56       
57       private Label createStatLabel(String title, String value, String color) {
58           VBox card = new VBox(5);
59           card.getStyleClass().add("card");
60           card.setPrefWidth(250);
61           card.setAlignment(Pos.CENTER);
62           
63           Label lblTitle = new Label(title);
64           lblTitle.setStyle("-fx-text-fill: #64748b; -fx-font-size: 14px;");
65           
66           Label lblValue = new Label("₹" + value);
67           lblValue.setStyle("-fx-text-fill: " + color + "; -fx-font-size: 24px; -fx-font-weight: bold;");
68           
69           card.getChildren().addAll(lblTitle, lblValue);
70           return lblValue; // Return value label reference to update later
71       }
72   
73       private VBox createExpenseManager() {
74           VBox root = new VBox(15);
75           root.setPadding(new Insets(10));
76           root.getStyleClass().add("card");
77           
78           Label title = new Label("Expense Manager");
79           title.setStyle("-fx-font-size: 18px; -fx-text-fill: #1e293b;");
80   
81           // Input Form
82           HBox form = new HBox(10);
83           form.setAlignment(Pos.CENTER_LEFT);
84           
85           ComboBox<String> cmbCat = new ComboBox<>();
86           cmbCat.setItems(FXCollections.observableArrayList("Rent", "Utilities", "Salary", "Maintenance", "Misc"));
87           cmbCat.setPromptText("Category");
88           
89           TextField txtDesc = new TextField(); txtDesc.setPromptText("Description");
90           TextField txtAmount = new TextField(); txtAmount.setPromptText("Amount");
91           
92           Button btnAdd = new Button("Add Expense");
93           btnAdd.getStyleClass().add("button-primary");
94           btnAdd.setOnAction(e -> {
95               try {
96                   double amt = Double.parseDouble(txtAmount.getText());
97                   String cat = cmbCat.getValue();
98                   if (cat == null) cat = "Misc";
99                   
100                  expenseDAO.addExpense(new Expense(0, cat, txtDesc.getText(), amt, null));
101                  refreshStats();
102                  loadExpenses();
103                  txtAmount.clear(); txtDesc.clear();
104              } catch (Exception ex) {
105                  new Alert(Alert.AlertType.ERROR, "Invalid Input").show();
106              }
107          });
108          
109          form.getChildren().addAll(cmbCat, txtDesc, txtAmount, btnAdd);
110  
111          // Table
112          expenseTable = new TableView<>();
113          
114          TableColumn<Expense, String> dateCol = new TableColumn<>("Date");
115          dateCol.setCellValueFactory(cell -> new javafx.beans.property.SimpleStringProperty(
116              cell.getValue().getDate().format(DateTimeFormatter.ofPattern("yyyy-MM-dd"))
117          ));
118          
119          TableColumn<Expense, String> catCol = new TableColumn<>("Category");
120          catCol.setCellValueFactory(new PropertyValueFactory<>("category"));
121          
122          TableColumn<Expense, String> descCol = new TableColumn<>("Description");
123          descCol.setCellValueFactory(new PropertyValueFactory<>("description"));
124          
125          TableColumn<Expense, Double> amtCol = new TableColumn<>("Amount");
126          amtCol.setCellValueFactory(new PropertyValueFactory<>("amount"));
127          
128          expenseTable.getColumns().addAll(dateCol, catCol, descCol, amtCol);
129          expenseTable.setColumnResizePolicy(TableView.CONSTRAINED_RESIZE_POLICY);
130          
131          loadExpenses();
132          
133          root.getChildren().addAll(title, form, expenseTable);
134          VBox.setVgrow(expenseTable, Priority.ALWAYS);
135          return root;
136      }
137  
138      private void refreshStats() {
139          double revenue = financeDAO.getTotalSales();
140          double cogs = financeDAO.getTotalPurchaseCosts();
141          double expenses = financeDAO.getTotalExpenses();
142          double profit = revenue - (cogs + expenses);
143          
144          lblRevenue.setText(String.format("₹%.2f", revenue));
145          lblCOGS.setText(String.format("₹%.2f", cogs));
146          lblExpenses.setText(String.format("₹%.2f", expenses));
147          lblProfit.setText(String.format("₹%.2f", profit));
148      }
149      
150      private void loadExpenses() {
151          try {
152              expenseTable.setItems(FXCollections.observableArrayList(expenseDAO.getAllExpenses()));
153          } catch (Exception e) { e.printStackTrace(); }
154      }
