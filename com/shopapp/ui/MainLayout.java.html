1    package com.shopapp.ui;
2    
3    import com.shopapp.Main;
4    import com.shopapp.models.User;
5    import javafx.animation.FadeTransition;
6    import javafx.scene.control.Button;
7    import javafx.scene.control.Label;
8    import javafx.scene.layout.BorderPane;
9    import javafx.scene.layout.Priority;
10   import javafx.scene.layout.Region;
11   import javafx.scene.layout.StackPane;
12   import javafx.scene.layout.VBox;
13   import javafx.stage.Stage;
14   import javafx.util.Duration;
15   
16   public class MainLayout extends BorderPane {
17   
18       private final StackPane contentArea;
19       private final User currentUser;
20       private Button activeNavButton = null;
21   
22       public MainLayout(User user) {
23           this.currentUser = user;
24           
25           // Create Sidebar
26           VBox sidebar = new VBox();
27           sidebar.getStyleClass().add("sidebar");
28           sidebar.setPrefWidth(260);
29   
30           Label title = new Label("Shop Manager");
31           title.getStyleClass().add("sidebar-title");
32           
33           Label userLabel = new Label("User: " + user.getUsername());
34           userLabel.setStyle("-fx-text-fill: #64748b; -fx-font-size: 12px; -fx-padding: 0 0 20 20;");
35   
36           Button btnDashboard = createNavButton("Dashboard");
37           Button btnBilling = createNavButton("Billing");
38           Button btnInventory = createNavButton("Inventory");
39           Button btnSuppliers = createNavButton("Suppliers");
40           Button btnOrders = createNavButton("Purchase Orders");
41           Button btnCustomers = createNavButton("Customers");
42           Button btnReports = createNavButton("Reports");
43   
44           sidebar.getChildren().add(title);
45           sidebar.getChildren().add(userLabel);
46           
47           sidebar.getChildren().add(btnDashboard);
48           sidebar.getChildren().add(btnBilling);
49           sidebar.getChildren().add(btnCustomers); // Available to all, or restrict? Let's make it available.
50           
51           // RBAC Logic
52           if (!"CASHIER".equalsIgnoreCase(user.getRole())) {
53               sidebar.getChildren().add(btnInventory);
54               sidebar.getChildren().add(btnSuppliers);
55               sidebar.getChildren().add(btnOrders); // Add PO Button
56           }
57           
58           if ("ADMIN".equalsIgnoreCase(user.getRole())) {
59               sidebar.getChildren().add(btnReports);
60           }
61   
62           Region spacer = new Region();
63           VBox.setVgrow(spacer, Priority.ALWAYS);
64           Button btnLogout = createNavButton("Logout");
65           btnLogout.setStyle("-fx-text-fill: #ef4444;");
66           btnLogout.setOnAction(e -> handleLogout());
67           sidebar.getChildren().addAll(spacer, btnLogout);
68   
69           // Create Content Area
70           contentArea = new StackPane();
71           contentArea.getStyleClass().add("content-area");
72           
73           // Assemble Layout
74           setLeft(sidebar);
75           setCenter(contentArea);
76           
77           // Navigation Logic
78           btnDashboard.setOnAction(e -> {
79               setActiveButton(btnDashboard);
80               setView(new DashboardView(target -> handleDashboardNavigation(target, btnInventory)));
81           });
82           
83           btnBilling.setOnAction(e -> {
84               setActiveButton(btnBilling);
85               setView(new BillingView());
86           });
87           
88           btnInventory.setOnAction(e -> {
89               setActiveButton(btnInventory);
90               setView(new InventoryView(false));
91           });
92           
93           btnSuppliers.setOnAction(e -> {
94               setActiveButton(btnSuppliers);
95               setView(new SuppliersView());
96           });
97           
98           btnOrders.setOnAction(e -> {
99               setActiveButton(btnOrders);
100              setView(new PurchaseOrderView());
101          });
102          
103          btnCustomers.setOnAction(e -> {
104              setActiveButton(btnCustomers);
105              setView(new CustomersView());
106          });
107          
108          btnReports.setOnAction(e -> {
109              setActiveButton(btnReports);
110              setView(new FinanceView());
111          });
112          
113          // Default View: Dashboard
114          btnDashboard.fire();
115      }
116      
117      // Handle navigation from Dashboard
118      private void handleDashboardNavigation(String target, Button inventoryBtn) {
119          switch (target) {
120              case "INVENTORY":
121                  if (inventoryBtn.getParent() != null) inventoryBtn.fire(); 
122                  break;
123              case "INVENTORY_LOW":
124                  if (inventoryBtn.getParent() != null) {
125                      setActiveButton(inventoryBtn);
126                      setView(new InventoryView(true));
127                  }
128                  break;
129              case "REPORTS":
130                   setView(new Label("Coming Soon: Detailed Reports"));
131                  break;
132          }
133      }
134      
135      private void handleLogout() {
136          Stage stage = (Stage) getScene().getWindow();
137          stage.close();
138          try {
139              new Main().start(new Stage());
140          } catch (Exception e) { e.printStackTrace(); }
141      }
142  
143      private Button createNavButton(String text) {
144          Button btn = new Button(text);
145          btn.getStyleClass().add("nav-button");
146          btn.setMaxWidth(Double.MAX_VALUE);
147          return btn;
148      }
149  
150      private void setActiveButton(Button btn) {
151          if (activeNavButton != null) {
152              activeNavButton.getStyleClass().remove("active");
153          }
154          activeNavButton = btn;
155          activeNavButton.getStyleClass().add("active");
156      }
157  
158      private void setView(javafx.scene.Node node) {
159          contentArea.getChildren().clear();
160          node.setOpacity(0);
161          contentArea.getChildren().add(node);
162          
163          FadeTransition ft = new FadeTransition(Duration.millis(300), node);
164          ft.setFromValue(0);
165          ft.setToValue(1);
166          ft.play();
167      }
168  }
